\باب{حر ناگزر تخمین}\شناخت{باب_حرارت_نا_گزر_تخمین}

\حصہ{مسئلہ حر ناگزر}
\جزوحصہ{حر ناگزر عمل} \شناخت{حصہ_حر_حرناگزر_عمل}
فرض کریں ایک کامل   رقاص   انتصابی سطح  میں بغیر کسی رگڑ یا ہوائی مزاحمت کے آگے پیچھے ارتعاش کرتا ہے۔ اگر آپ اس   رقاص   کو  جھٹکے سے ہلائیں تو یہ افراتفری کے ساتھ  حرکت کرنے لگے گا،  لیکن اگر آپ بغیر جھٹکا دیے    رقاص   کو آہستہ آہستہ ایک مقام سے دوسرے  مقام منتقل کریں (شکل \حوالہ{شکل_حرارت_نا_گزر_آہستہ_منتقلی})    تو   یہ    اسی سطح  (یا اس کی  متوازی سطح)  میں شائستگی اور روانی سے،  اسی حیطہ کے ساتھ جھلولتا رہے گا۔\ترچھا{ بیرونی کیفیت  کی بہت آہستہ  تبدیلی}  ہی \اصطلاح{ حر ناگزر}\فرہنگ{حر ناگزر}\حاشیہب{adiabatic}\فرہنگ{adiabatic} عمل کی پہچان ہے۔ دھیان رہے کہ یہاں دو مختلف   وقتوں کی بات کی جا رہی ہے:  نظام کی اپنی  حرکت (  جو یہاں   رقاص   کے  ارتعاش کا دوری عرصہ ہوگا)   کو ظاہر کرنے والا\قول{ اندرونی}  وقت \عددی{T_i}،  اور نظام  کی مقادیر معلوم میں  نمایاں تبدیلی( مثلاً،  لرزتے ہوئے چبوترے  پر نصب   رقاص   کی صورت میں چبوترے کی لرزش کا دوری عرصہ)  کو ظاہر کرنے والا \قول{ بیرونی}  وقت \عددی{T_e}۔ حر ناگزر عمل میں \عددی{T_e \gg T_i} ہوگا۔

 حر ناگزر عمل کے تجزیے  کی بنیادی حکمت عملی یہ  ہے کہ پہلے بیرونی  مقادیر معلوم کو غیر متغیر رکھتے ہوئے مسئلہ حل کیا جاتا ہے،  اور حساب کے بالکل آخر میں انہیں  (بہت آہستہ)   وقت کے ساتھ تبدیل ہونے کی اجازت دی جاتی ہے۔ مثال کے طور پر، مقررہ لمبائی \عددی{L} کے   رقاص  کا کلاسیکی دوری عرصہ \عددی{2 \pi \sqrt{L/g}} ہوگا؛  اب اگر لمبائی آہستہ آہستہ  \ترچھا{تبدیل ہو}،  تو دوری عرصہ  بظاہر  \عددی{2 \pi \sqrt{L(t)/g}} ہوگا۔    ہائیڈروجن سالمہ  (حصہ \حوالہ{حصہ_تغیری_اصول_ہائیڈروجن_بارداریہ}) پر تبصرہ کے دوران زیادہ باریک بین مثال پیش کی گئی۔ ہم نے  مراکزہ کو \ترچھا{ ساکن} تصور کرتے ہوئے آغاز  کیا،   اور  ان کے بیچ فاصلہ \عددی{ R} کی صورت میں الیکٹران کی حرکت کے لئے حل کیا۔  نظام کی زمینی حال توانائی کو \عددی{ R} کے تفاعل کی صورت میں دریافت کرنے کے بعد،  ہم نے توازنی فاصلہ معلوم کرکے ترسیم کے  انحنا سے مراکزہ کی لرزش کا تعدد حاصل کیا (سوال \حوالہ{سوال_تغیری_توازنی_فاصلہ})۔طبیعیات سالمہ میں اس ترکیب کو ( جس میں ساکن مراکزہ سے آغاز کرتے ہوئے،  الیکٹرانی تفاعلات موج کا حساب کر کے،  ان سے نسبتاً سست رفتار مراکزہ کے مقامات اور حرکت کے بارے میں معلومات حاصل کرنے کو)   \اصطلاح{بارن و اوپن ہائیمر تخمین}\فرہنگ{بارن و اوپن ہائیمر تخمین}\حاشیہب{Born-Oppenheimer approximation}\فرہنگ{Born-Oppenheimer approximation}  کہتے ہیں۔
 
  کوانٹائی میکانیات میں،\اصطلاح{ حر ناگزر تخمین}\فرہنگ{حر ناگزر!تخمین}\حاشیہب{adiabatic approximation}\فرہنگ{adiabatic!approximation} کے بنیادی تصور کو ایک مسئلہ کے روپ میں پیش کیا جا سکتا ہے۔ فرض کریں ہیملٹنی ابتدائی روپ \عددی{H^i} سے بہت آہستہ  تبدیل ہو کر کسی اختتامی روپ  \عددی{H^f} تک پہنچتی  ہے۔ \اصطلاح{ مسئلہ حر ناگزر}\فرہنگ{حر ناگزر!مسئلہ}\حاشیہب{adiabatic theorem}\فرہنگ{adiabatic!theorem} کہتا ہے کہ اگر ذرہ  ابتدائی طور پر \عددی{H^i} کے \عددی{n} وی امتیازی حال میں پایا جاتا ہو،  تو  ( زیر مساوات شروڈنگر) یہ   \عددی{H^f} کے  \عددی{ n} وی امتیازی حال میں منتقل ہوگا۔ ( میں یہاں فرض کرتا ہوں کہ \عددی{H^i} سے \عددی{H^f} تک تحویل کے دوران،  طیف غیر مسلسل اور غیر انحطاطی ہے،  لہٰذا   حالات کی ترتیب میں  کوئی شبہ نہیں پایا جائے گا؛ امتیازی تفاعلات پر نظر رکھنے کی کوئی ترکیب وضع کرنے سے ان شرائط کو نرم بنایا جا سکتا ہے،  لیکن میں یہاں ایسا نہیں کروں گا۔)

 مثال کے طور پر،  ہم لامتناہی چوکور کنویں میں ایک ذرے   کو زمینی حال:
\begin{align}
\psi^i (x) = \sqrt{\frac{2}{a}} \sin \big ( \frac{\pi}{a} x \big )
\end{align}
 میں تیار کرتے ہیں (شکل \حوالہ{شکل_حرارت_نا_گزر_ابتدائی_حال_کنواں_دیوار}-ا)۔  اب دائیں  دیوار کو بہت آہستہ  مقام \عددی{2a} پر منتقل کیا جاتا ہے؛  مسئلہ حر ناگزر کے تحت ( ماسوائے ہیّتی  جزو ضربی  کے)  یہ ذرہ توسیع شدہ کنویں کے زمینی حال:
\begin{align}
\psi^f (x) = \sqrt{\frac{1}{a}} \sin \big ( \frac{\pi}{2a} x \big )
\end{align}
 میں منتقل ہوگا  (شکل  \حوالہ{شکل_حرارت_نا_گزر_ابتدائی_حال_کنواں_دیوار}-ب)۔ دھیان رہے کہ  ہم ہیملٹنی میں   چھوٹی تبدیلی     (نظریہ اضطراب کی طرح)  کی بات نہیں کر رہے ہیں؛ ہاں تبدیلی بہت بڑی ہے۔  فقط اتنا ضروری  ہے کہ تبدیلی \ترچھا{  آہستہ}   رونما ہو۔ یہاں توانائی کی بقا نہیں ہوگی: جو بھی دیوار کو حرکت دے رہا ہے،  نظام سے توانائی حاصل کرے گا ، جیسا  کہ  گاڑی کے  انجن کے سلنڈر   میں آہستہ آہستہ پھیلتا ہوا گیس بوکا کو توانائی فراہم کرتا ہے۔ اس کے برعکس،  کنویں کی \ترچھا{ اچانک} وسعت کی صورت میں   حال \عددی{\psi^i (x)} ہی رہتا ہے  (شکل  \حوالہ{شکل_حرارت_نا_گزر_ابتدائی_حال_کنواں_دیوار}-ج)،    جو نئی  ہیملٹنی کے امتیازی حالات کا  پیچیدہ خطی جوڑ ہوگا  (سوال \حوالہ{سوال_غیر_تابع_پیچیدہ_خطی_جوڑ})۔ یہاں توانائی  ( کم  از کم،  اس کی توقعاتی قیمت) کی   بقا ہوگی؛  جیسا اچانک رکاوٹ ہٹانے سے خلاء میں گیس کی آزادانہ پھیلانے  سے کوئی کام نہیں ہوتا۔


\ابتدا{سوال}\شناخت{سوال_حر_وقت_کا_تابع_مقدار_معلوم}
ایک لامتناہی چوکور کنواں،  جس کی دائیں دیوار ایک مستقل سمتی رفتار \عددی{v} سے حرکت کرتے ہوئے کنویں کو وسیع بناتی ہے،  کو  \ترچھا{ٹھیک ٹھیک}  حل کرنا ممکن ہے۔  اس کے حلوں کا مکمل سلسلہ درج ذیل ہوگا 
\begin{align}\label{مساوات_حر_مکمل_حل_چوکور_کنواں}
\Phi_n (x, t) \equiv \sqrt{\frac{2}{\omega}} \sin \big ( \frac{n \pi}{\omega} x \big ) e^{i(mvx^2 -2E_n^i at)/2\hslash \omega}
\end{align}
جہاں \عددی{w(t) \equiv a + vt } کنویں کی ( لمحاتی)  چوڑائی اور \عددی{E_n^i \equiv n^2 \pi^2 \hslash^2 /2ma^2}  ( چوڑائی \عددی{ a}) کے \ترچھا{ اصل}  کنویں کی \عددی{ n} ویں اجازتی   توانائی  ہے۔ عمومی حل ان \عددی{\Phi} کا  خطی جوڑ:
\begin{align}
\Psi (x, t) = \sum_{n = 1}^{\infty} c_n \Phi_n (x, t)
\end{align}
ہوگا،  جہاں عددی سر \عددی{c_n} وقت \عددی{ t} کے تابع نہیں ہوں گے۔
\begin{enumerate}[a.]
\item
دیکھیں آیا تابع وقت مساوات شروڈنگر بمع مناسب سرحدی شرائط کو مساوات  \حوالہ{مساوات_حر_مکمل_حل_چوکور_کنواں}   مطمئن کرتی ہے۔
\item
فرض کریں اصل کنویں کے  زمینی حال میں ایک ذرہ آغاز \عددی{(t=0)}  کرتا ہے۔
\begin{align*}
\Psi (x, 0) = \sqrt{\frac{2}{a}} \sin \big ( \frac{\pi}{a} x \big )
\end{align*}
دکھائیں کے  توسیعی   عددی سروں کو درج ذیل روپ میں لکھا جا سکتا ہے 
\begin{align}
c_n = \frac{2}{\pi} \int_0^{\pi} e^{-i\alpha z^2} \sin(nz) \sin(z) \dif z
\end{align}
جہاں \عددی{\alpha \equiv mva/2 \pi^2 \hslash} کنویں کے پھیلنے کی رفتار کی  بے بُعدی پیمائش ہے۔ ( بدقسمتی سے اس تکمل کی قیمت  بنیادی تفاعلات کی صورت میں حاصل نہیں کی جا سکتی۔) 
\item
فرض کریں ہم کنویں کو ابتدائی چوڑائی کے دگنی چوڑائی تک پھیلنے دیتے ہیں،  یوں  \قول{بیرونی}  وقت \عددی{w(T_e) = 2a} ہوگا۔  (ابتدائی)  زمینی حال کے تابع وقت قوت نما  جزو ضربی کا دوری عرصہ ،   \قول{اندرونی} وقت ہوگا۔ وقت \عددی{T_e} اور \عددی{T_i} کا  تعین کرکے،  دکھائیں  کہ  حر ناگزر   طریق  سے مراد \عددی{\alpha \ll 1} ہوگا،  لہٰذا  تکمل کے
 دائرہ کار پر \عددی{e^{-i \alpha z^2} \cong 1} ہوگا۔ اس کو استعمال کرتے ہوئے  توسیعی  عددی سر \عددی{c_n} کا  تعین کریں۔  حال \عددی{\Psi (x, t)} تیار کرکے تصدیق کریں کہ یہ مسئلہ حر ناگزر کے مطابق ہے۔
\item
دکھائیں کہ  \عددی{\Psi (x, t)} میں    ہیّتی جزو ضربی  کو درج ذیل روپ میں لکھا جا سکتا ہے 
\begin{align}
\theta (t) = - \frac{1}{\hslash} \int_0^1 E_1 (t') \dif t'
\end{align}
جہاں لمحہ \عددی{ t} پر \ترچھا{ لمحاتی}  امتیازی قدر \عددی{E_n (t) \equiv n^2 \pi^2 \hslash^2 /2m \omega^2} ہوگا۔ اس نتیجہ پر تبصرہ کریں۔
\end{enumerate}
\انتہا{سوال}
%================
\جزوحصہ{مسئلہ حر ناگزر کا ثبوت} \شناخت{حصہ_حر_مسئلہ_حرناگزر_کا_ثبوت}
مسئلہ حر ناگزر بظاہر معقول نظر آتا ہے،  اور اسے باآسانی بیان کیا جا سکتا ہے،  تاہم اس کو ثابت کرنا اتنا آسان نہیں۔  \ترچھا{غیر تابع وقت} ہیملٹنی کی صورت میں،  ایک ذرہ جو \عددی{n} وی امتیازی حال،\حاشیہد{میں مقام (یا چکر، وغیرہ)  کا ذکر  نہیں کروں گا ، چونکہ  اس دلیل میں تابعیت وقت کی بات کی جا رہی ہے۔} \عددی{\psi_n}،   میں آغاز کرتا ہے ،
\begin{align}
H \psi_n = E_n \psi_n
\end{align}
 ہیّتی  جزو ضربی اپنانے کے علاوہ اسی \عددی{n} وی امتیازی حال:
\begin{align}
\Psi_n (t) = \psi_n e^{-i E_n t/ \hslash}
\end{align}
میں رہتا ہے۔  اگر ہیملٹنی  \ترچھا{وقت} کے ساتھ تبدیل ہوتا ہو،  تب امتیازی تفاعلات اور امتیازی اقدار بھی تابع وقت ہوں گے:
\begin{align}\label{مساوات_حر_تابع_وقت_ہیملٹنی_توانائی}
H(t) \psi_n (t) = E_n (t) \psi_n (t)
\end{align}
لیکن اب بھی ( کسی ایک مخصوص لمحہ پر)  یہ معیاری  عمودی سلسلہ: 
\begin{align}\label{مساوات_حر_لمحاتی_معیاری_عمودی}
\langle \psi_n (t) | \psi_m (t) \rangle \delta_{nm} 
\end{align}
دیں گے، اور یہ     مکمل ہیں،  لہٰذا تابع وقت مساوات شروڈنگر 
\begin{align}\label{مساوات_حر_مساوات_شروڈنگر_الف}
i \hslash \frac{\partial}{\partial t} \Psi (t) = H (t) \Psi (t)
\end{align}
کے عمومی حل کو ان کا خطی جوڑ:
\begin{align}\label{مساوات_حر_مساوات_شروڈنگر_ب}
\Psi (t) = \sum_n c_n (t) \psi_n (t) e^{i \theta_n (t)}
\end{align}
لکھا جا سکتا ہے، جہاں 
\begin{align}\label{مساوات_حر_مساوات_شروڈنگر_پ}
\theta_n (t) \equiv  - \frac{1}{\hslash} \int_0^t E_n (t') \dif t'
\end{align}
وقت کے ساتھ تبدیل ہوتے ہوئے \عددی{E_n} کی صورت میں \قول{معیاری}   ہیّتی جزو ضربی کو عمومیت دیتا ہے۔ (  ہمیشہ کی طرح میں اس کو  عددی سر \عددی{c_n (t)} میں ضم  کر سکتا تھا،  لیکن غیر تابع وقت ہیملٹنی کی صورت میں بھی یہ پایا جاتا ہے،  لہٰذا  تابعیت  وقت کے اس حصے  کو صریحاً  لکھنا موزوں ہوگا۔) 

 مساوات  \حوالہ{مساوات_حر_مساوات_شروڈنگر_ب}  کو مساوات  \حوالہ{مساوات_حر_مساوات_شروڈنگر_الف}  میں پر کرنے سے
\begin{align}
i \hslash \sum_n [\dot{c}_n \psi_n + c_n \dot{\psi}_n + i c_n \psi_n \dot{\theta}_n] e^{i \theta_n} = \sum_n c_n (H \psi_n) e^{i \theta_n} 
\end{align}
حاصل ہو گا  (جہاں وقت کے لحاظ سے تفرق کو نقطہ  سے ظاہر کیا گیا ہے)۔  مساوات  \حوالہ{مساوات_حر_تابع_وقت_ہیملٹنی_توانائی}  اور  مساوات \حوالہ{مساوات_حر_مساوات_شروڈنگر_پ}  کی بنا پر آخری دو اجزاء کٹتے ہیں،  لہٰذا درج ذیل باقی رہتا ہے۔ 
\begin{align}
\sum_n \dot{c}_n \psi_n e^{i \theta_n} = - \sum_n c_n \dot{\psi}_n e^{i \theta_n}
\end{align}
اس کا \عددی{\psi_m} کے ساتھ اندرونی ضرب لے کر،  لمحاتی امتیازی تفاعلات کی معیاری  عمودیت (مساوات \حوالہ{مساوات_حر_لمحاتی_معیاری_عمودی}) بروئے کار لاتے ہوئے 
\begin{align*}
\sum_n \dot{c}_n \delta_{mn} e^{i \theta_n} = - \sum_n c_n \langle \psi_m | \dot{\psi}_n \rangle e^{i \theta_n}
\end{align*}
یا درج ذیل  حاصل ہوگا۔ 
\begin{align}\label{مساوات_حر_تفرق_عددی_سر}
\dot{c}_m (t) = - \sum_n c_n \langle \psi_m | \dot{\psi}_n \rangle e^{i(\theta_n - \theta_m)}
\end{align}

اب مساوات \حوالہ{مساوات_حر_تابع_وقت_ہیملٹنی_توانائی}  کا وقت کے ساتھ تفرق لیتے ہیں 
\begin{align*}
\dot{H} \psi_n + H \dot{\psi}_n = \dot{E}_n \psi_n + E_n \dot{\psi}_n
\end{align*}
اور  یوں (دوبارہ  \عددی{\psi_m} کے ساتھ اندرونی ضرب لے کر)  درج ذیل حاصل  ہوگا۔ 
\begin{align}
\langle \psi_m | \dot{H} | \psi_n \rangle + \langle \psi_m | H | \dot{\psi}_n \rangle = \dot{E}_n \delta_{mn} + E_n \langle \psi_m | \dot{\psi}_n \rangle
\end{align} 
ہم \عددی{ H} کے ہرمشی  پن  سے فائدہ اٹھاتے ہوئے \عددی{\langle \psi_m | H | \dot{\psi}_n \rangle = E_m \langle \psi_m | \dot{\psi}_n \rangle} لکھتے ہیں،  یوں  \عددی{n \ne m} کی صورت  میں درج ذیل ہوگا۔
\begin{align}\label{مساوات_حر_نابرابر_صورت}
\langle \psi_m | \dot{H} | \psi_n \rangle = (E_n - E_m) \langle \psi_m | \dot{\psi}_n \rangle
\end{align}
(یہ جانتے ہوئے کہ  توانائیاں غیر انحطاطی ہیں)  مساوات  \حوالہ{مساوات_حر_نابرابر_صورت}    کو مساوات  \حوالہ{مساوات_حر_تفرق_عددی_سر}  میں پُر کر کے درج ذیل اخذ ہوگا۔
\begin{align}
\dot{c}_m (t) = - c_m \langle \psi_m | \dot{\psi}_m \rangle - \sum_{n \ne m} c_n \frac{\langle \psi_m | \dot{H} | \psi_n \rangle}{E_n - E_m} e ^{(- i / \hslash) \int_0^t [E_n(t') - E_m(t')] \dif t' }
\end{align}

یہ   \اصطلاح{ٹھیک ٹھیک}  نتیجہ ہے۔ اب حر ناگزر تخمین کی باری آتی ہے: فرض کریں \عددی{\dot{H}} نہایت چھوٹا ہے، اور  دوسرے  جزو  کو نظر انداز کرتے ہوئے \حاشیہد{اس قدم کی  پختہ وجہ تلاش کرنا اتنا آسان نہیں۔ }
\begin{align}
\dot{c}_m (t) = - c_m \langle \psi_m | \dot{\psi}_m \rangle
\end{align} 
ہوگا، جس کا حل 
\begin{align}\label{مساوات_حر_سی_ایم}
c_m (t) = c_m (0) e^{i \gamma_m (t)}
\end{align}
ہے، جہاں درج ذیل ہوگا۔\حاشیہد{چونکہ \عددی{\psi_m} کی معمول زنی  میں
\عددی{(\dif/\dif t)\langle\psi_m|\psi_m\rangle=\langle\psi_m|\dot{\psi}_m\rangle+\langle\dot{\psi}_m|\psi_m\rangle=2(\langle\psi_m|\dot{\psi}_m)_{\text{\RL{حقیقی}}}=0} شامل ہے لہٰذا \عددی{\gamma} حقیقی ہو گا۔
}
\begin{align}
\gamma_m (t) \equiv  i \int_0^t \big\langle \psi_m (t') \big| \frac{\partial}{\partial t'} \psi_m (t') \big \rangle \dif t'
\end{align}
بالخصوص، اگر ذرہ \عددی{ n} وی امتیازی حال  (یعنی \عددی{m \ne n} کیلئے \عددی{c_n (0) = 1} اور \عددی{c_m (0) = 0} ہو)  سے آغاز کرے،  تب  (مساوات \حوالہ{مساوات_حر_مساوات_شروڈنگر_ب}) 
\begin{align}\label{مساوات_حر_وہیں_رہتا_ہے}
\Psi_n (t) = e^{i \theta_n (t)} e^{i \gamma_n (t)} \psi_n (t)
\end{align}
ہوگا، لہٰذا  (چند  ہیّتی جزو ضربی  حاصل کرنے کے علاوہ)  یہ ذرہ   (ارتقائی  ہیملٹنی کے)  \عددی{ n} وی امتیازی حال میں ہی رہے گا۔
%=========================

\ابتدا{مثال}\شناخت{مثال_حر_مخصوص_صورت}
فرض کریں  مقناطیسی میدان میں مبدا  پر ( کمیت \عددی{ m} اور بار \عددی{-e} کا )  ساکن  الیکٹران  پایا جاتا ہے۔ اس مقناطیسی میدان کی  \ترچھا{مقدار}  \عددی{(B_0)}  مستقل ہے،  جبکہ اس کا  \ترچھا{رخ}  \عددی{z} محور کے گرد،    مقررہ زاویائی  سمتی رفتار \عددی{\omega} کے ساتھ،   مخروطی سطح پر رہتے ہوئے گھومتا ہے۔ محور \عددی{z} کے ساتھ  مخروط کا اندرونی
 زاویہ \عددی{\alpha} ہے  (شکل \حوالہ{شکل_حرارت_نا_گزر_مخروطی_راہ_جھاڑنا})۔
\begin{align}\label{مساوات_حر_مخروط_مقناطیسی_میدان}
\kvec{B} (t) = B_0 [\sin(\alpha) \cos(\omega t) \ai + \sin(\alpha) \sin(\omega t) \aj + \cos \alpha \ak]
\end{align}

اس کی ہیملٹنی (مساوات  \حوالہ{مساوات_تین_ابعادی_جفت_قطب_ہیملٹنی})   درج ذیل ہوگی
\begin{gather}
\begin{aligned}\label{مساوات_حر_مخصوص_ہیملٹنی}
H(t) &= \frac{e}{m} \kvec{B} \cdot \kvec{S} = \frac{e \hslash B_0}{2m} [\sin \alpha \cos(\omega t) \sigma_x + \sin \alpha \sin(\omega t) \sigma_y + \cos \alpha \sigma_z] \\
&= \frac{\hslash \omega_1}{2}
\begin{pmatrix}
\cos \alpha& e^{-i\omega t}\sin \alpha\\[0.5em]
e^{i\omega t}\sin\alpha& -\cos\alpha
\end{pmatrix} 
\end{aligned}
\end{gather}
جہاں \عددی{\omega_1} درج ذیل ہے۔
\begin{align}
\omega_1 \equiv \frac{e B_0}{m}
\end{align}
ہیملٹنی \عددی{H(t)} کے معمول شدہ امتیازی چکر کار  \عددی{\chi_+} اور \عددی{\chi_-} درج ذیل ہیں
\begin{align}
\chi_+ (t) &= 
\begin{pmatrix}
\cos(\alpha/2) \\[0.5em]
e^{i \omega t} \sin(\alpha/2)
\end{pmatrix}\\
\chi_{-} (t) &= 
\begin{pmatrix}
e^{-i \omega t} \sin(\alpha/2) \\[0.5em]
- \cos(\alpha/2)
\end{pmatrix}
\end{align}
جو \عددی{\kvec{B} (t)} کے لمحاتی رخ کے ساتھ، بالترتیب،  ہم چکر اور خلاف چکر کو ظاہر کرتے ہیں (سوال  \حوالہ{سوال_تین_بعدی_قالب_ایس_زاویائی_معیار_حرکت}  دیکھیں)۔  ان کی مطابقتی امتیازی اقدار درج ذیل ہوں گی۔
\begin{align}\label{مساوات_حر_حرکی_ہیّت_توانائی}
E \pm = \pm \frac{\hslash \omega_1}{2}
\end{align}

فرض کریں \عددی{\kvec{B} (0)} کی  ہم راہ،    الیکٹران ہم چکر:
\begin{align}
\chi (0) = 
\begin{pmatrix} 
\cos(\alpha/2) \\[0.5em]
\sin(\alpha/2)
\end{pmatrix}
\end{align}
  صورت سے آغاز کرتا ہے۔\حاشیہد{یہ بنیادی طور پر سوال \حوالہ{سوال_تابع_مضطرب_گمک} ہی ہے، البتہ یہاں الیکٹران \عددی{\kvec{B}} کی ہم راہ، ہم چکر سے آغاز کرتا ہے، جبکہ سوال  \حوالہ{سوال_تابع_مضطرب_گمک}-د میں یہ \عددی{z} محور کی ہم راہ، ہم چکر سے آغاز کرتا ہے۔}  تابع وقت مساوات شروڈنگر کا  ٹھیک ٹھیک  حل درج ذیل ہوگا  (سوال  \حوالہ{سوال_حر_ٹھیک_حل})
\begin{align}\label{مساوات_حر_حل_مخصوص_ہیملٹنی}
\chi (t) = 
\begin{pmatrix}
[\cos(\lambda t/2) - i \frac{(\omega_1 - \omega)}{\lambda} \sin(\lambda t/2)] \cos(\alpha/2) e^{- i \omega t/2} \\[0.5em]
[\cos(\lambda t/2) - i \frac{(\omega_1 + \omega)}{\lambda} \sin(\lambda t/2)] \sin(\alpha/2) e^{+ i \omega t/2}
\end{pmatrix}
\end{align}
جہاں \عددی{\lambda} درج ذیل  ہے۔
\begin{align}
\lambda \equiv \sqrt{\omega^2 + \omega_1^2 - 2 \omega \omega_1 \cos \alpha}
\end{align}
اس حل  کو  \عددی{\chi_{+}} اور \عددی{\chi_{-}} کا خطی  جوڑ  لکھا جا سکتا ہے۔
\begin{multline}\label{مساوات_حر_حل_خطی_جوڑ}
\chi (t) = \big [ \cos\big ( \frac{\lambda t}{2} \big ) - i \frac{(\omega_1 - \omega \cos \alpha)}{\lambda} \sin\big ( \frac{\lambda t}{2} \big ) \big ] e^{- i \omega t/2} \chi_{+} (t) \\
+ i \big [ \frac{\omega}{\lambda} \sin \alpha \sin \big ( \frac{\lambda t}{2} \big ) \big ] e^{+ i \omega t/2} \chi_- (t)
\end{multline}
ظاہر ہے کہ  (\عددی{\kvec{B}} کے موجودہ رخ کے لحاظ سے)  خلاف چکر  تحویل کا ٹھیک ٹھیک احتمال درج ذیل ہوگا۔ 
\begin{align}\label{مساوات_حر_تحویلی_احتمال}
\abs{\langle \chi (t) | \chi_{-} (t) \rangle}^2 = \big [ \frac{\omega}{\lambda} \sin \alpha \sin \big ( \frac{\lambda t}{2} \big ) \big ]^2
\end{align}
مسئلہ حر ناگزر کہتا ہے کہ \عددی{T_e \gg T_i} کی تحدیدی صورت میں تحویلی احتمال صفر کو پہنچے گا،  جہاں ہیملٹنی میں تبدیلی کو درکار امتیازی وقت \عددی{T_e} ہے  (جو موجودہ صورت میں \عددی{1/\omega} ہوگا)،  اور تفاعل موج میں تبدیلی کے لیے درکار امتیازی وقت \عددی{T_i} ہے ( جو موجودہ صورت 
میں \عددی{\hslash / (E_{+} - E_{-}) = 1/\omega_1)} ہوگا)۔  یوں حر ناگزر تخمین سے مراد \عددی{\omega \ll \omega_1} ہے؛ (غیر مضطرب) تفاعلات موج کی ہیّت  کے لحاظ سے میدان آہستہ گھومتا ہے۔  حر ناگزر  طریق میں   \عددی{\lambda \cong \omega_1} لہٰذا 
\begin{align}
\abs{\langle \chi (t) | \chi_{-} (t) \rangle}^2 \cong \big [ \frac{\omega}{\omega_1} \sin \alpha \sin \big ( \frac{\lambda t}{2} \big ) \big ]^2 \to 0
\end{align}

  اگر میں اس کامل  رقاص  کو شمالی قطب پر لے جا کر،   مثلاً  صوابی شہر کے رخ،    جھولا دوں  (شکل \حوالہ{شکل_حرارت_نا_گزر_سطح_زمین_منتقل}) تو کیا ہو گا؟  فی الحال تصور کریں کہ دنیا گھوم نہیں رہی ہے۔  میں اس کو بہت آہستہ ( یعنی حر ناگزر طریقہ سے)  صوابی سے گزرتے  خط طول بلد پر چلتے ہوئے،   خط استوا تک پہنچتا ہوں۔ یہاں پہنچ کر یہ شمال و جنوب جھولے  رہا ہو گا۔ میں اس کو  خط استوا پر کچھ فاصلہ دور تک لے جاتا ہوں (  رقاص  ابھی بھی شمال و جنوب جھولتا ہے)۔  آخر میں  نئے  خط طول بلند پر چلتے ہوئے، میں رقاص کو   واپس شمالی قطب منتقل کرتا ہوں۔  آپ دیکھ سکتے ہیں کہ   رقاص  اب  اسی مستوی میں  نہیں جھولے گا جس سے اس نے آغاز کیا تھا؛   یقیناً،  نئے  اور پرانے   مستوی کے بیچ زاویہ \عددی{\Theta} پایا جاتا ہے،  جہاں جنوب کی طرف چلتے ہوئے اور شمال کی طرف چلتے ہوئے  خط طول بلند کے بیچ زاویہ \عددی{\Theta} ہے۔ 
  
    جس راہ پر میں  رقاص کو  اٹھا کر چلتا رہا،  وہ راہ ( زمین کے مرکز پر)   \اصطلاح{ٹھوس زاویہ}\فرہنگ{ٹھوس زاویہ}\حاشیہب{solid angle}\فرہنگ{solid angle}  ، \عددی{\Omega}،  بناتی ہے اور   \عددی{\Theta} اسی \عددی{(\Omega)} کے برابر ہے۔  یہ راہ شمالی نصف کرہ کا \عددی{\Theta /2\pi} حصہ گھیرتی ہے،  لہٰذا

 کرہ کی سطح پر  بند راہ پر چلتے ہوئے حر ناگزر منتقلی کی ایک مثال  \اصطلاح{فوقو    رقاص}\فرہنگ{فوقو رقاص}\حاشیہب{Foucault pendulum}\فرہنگ{Foucault pendulum}   ہے،  جہاں  رقاص  کو اٹھا کر چلنے کا کام مجھے نہیں  بلکہ زمین کے گھومنے کو  سونپا جاتا ہے۔  خط عرض بلد \عددی{\theta_0} درج ذیل ٹھوس زاویہ بناتا ہے  (شکل   \حوالہ{شکل_حرارت_نا_گزر_فوقو_رقاص_ایک_دن})۔ 
\begin{align}\label{مساوات_حر_ٹھوس_زاویہ}
\Omega = \int \sin \theta \dif \theta \dif \phi =\left.  2\pi (- \cos \theta)\right\vert_0^{\theta_0} = 2 \pi (1 - \cos \theta_0)
\end{align}
%



زمین کے لحاظ سے ( جو اس دوران \عددی{2\pi} زاویہ گھوم چکی  ہوگی)  فوقو    رقاص   کی روزانہ استقبالی حرکت  \عددی{2\pi \cos \theta_0} ہوگی؛ اس نتیجہ کو، عموماً،   گھومتی حوالہ چوکھٹ پر  \اصطلاح{کوریولس}\فرہنگ{کوریولس}\حاشیہب{Coriolis}\فرہنگ{Coriolis}  قوتوں کے اثر  سے حاصل کیا جاتا ہے، لیکن یہاں یہ خالصتاً  \ترچھا{ ہندسی}  مفہوم کا حامل ہے۔ 

 ایسا نظام جو بند راہ پر چلتے ہوئے   واپس ابتدائی  نقطہ  پہنچ کر اپنے  ابتدائی حال کو  نہیں لوٹتا  \اصطلاح{ گرگٹی}\فرہنگ{گرگٹی}\حاشیہب{nonholonomic}\فرہنگ{nonholonomic}  کہلاتا ہے۔(  یہاں ضروری نہیں کہ  راہ پر چلنے سے مراد \قول{حرکت دینا}  ہو؛  اس سے مراد صرف اتنا ہے کہ نظام کی مقدار معلوم قیمتوں کو یوں تبدیل کیا جاتا ہے کہ آخر کار ان کی قیمتیں وہی ہوں جو ابتدا میں تھیں۔)   گرگٹی  نظام  جگہ جگہ پائے جاتے ہیں؛  ایک لحاظ سے ہر چکر دار انجن  گرگٹی  ہے:  ہر ایک پھیرے  کے اختتام تک گاڑی آگے حرکت کر چکی ہوگی،  یا کوئی وزن اٹھایا گیا ہوگا،  وغیرہ۔ اس تصور کا اطلاق،    چھوٹے \اصطلاح{ رینالڈ عدد}\فرہنگ{رینالڈ عدد}\حاشیہب{Reynolds number}\فرہنگ{Reynolds number}   پر  سیال میں،  جرثوموں کی حرکت پر بھی کیا گیا ہے۔     اگلے حصہ میں میں  \ترچھا{گرگٹی   حر ناگزر عمل}   کی  \ترچھا{کوانٹائی میکانیات} پر غور کروں گا۔ ہم نے دیکھنا ہوگا کہ ہیملٹنی کی  مقدار معلوم مقداروں کو کسی بند راہ پر حر ناگزر پھیرا دینے سے اختتامی حال کس طرح ابتدائی حال سے مختلف ہوگا۔ 
%==========================


\جزوحصہ{ہندسی ہیّت}
میں نے حصہ  \حوالہ{حصہ_حر_مسئلہ_حرناگزر_کا_ثبوت}  میں دکھایا کہ  ایک ذرہ جو \عددی{H(0)} کے \عددی{n} وی امتیازی حال سے آغاز کرتا ہے،  حر ناگزر صورت  میں، تابع وقت ہیّتی جزو ضربی کے علاوہ،  \عددی{H(t)} کے  \عددی{n} وی امتیازی حال میں  رہتا ہے۔ بالخصوص،  اس کا تفاعل موج (مساوات  \حوالہ{مساوات_حر_وہیں_رہتا_ہے}):
\begin{align}
\Psi_n (t) = e^{i[\theta_n (t) + \gamma_n (t)]} \psi_n (t)
\end{align} 
ہو گا، جہاں 
\begin{align}\label{مساوات_حر_درست_روپ}
\theta_n (t) \equiv - \frac{1}{\hslash} \int_0^t E_n (t') \dif t'
\end{align}
 \اصطلاح{حرکی ہیّت}\فرہنگ{حرکی ہیّت}\حاشیہب{dynamic phase}\فرہنگ{dynamic phase} ہے ( جو وقت کے  تفاعل \عددی{E_n} کی صورت میں،   جزو ضربی \عددی{e^{(- i E_n t/\hslash)}} کو عمومیت دیتی ہے)،   اور درج ذیل \اصطلاح{ ہندسی ہیّت}\فرہنگ{ہندسی ہیّت}\حاشیہب{geometric phase}\فرہنگ{geometric phase} کہلاتی ہے۔
\begin{align}
\gamma_n(t)\equiv i\int_0^t \big\langle \psi_n(t')\big|\frac{\partial}{\partial t'}\psi_n(t')\big \rangle \dif t'
\end{align}

 چونکہ  ہیملٹنی میں کوئی ایسی مقدار معلوم \عددی{R(t)} پائی  جاتی  ہے جو وقت کے ساتھ تبدیل ہوتی  ہے، لہٰذا \عددی{\psi_n (t)} وقت \عددی{ t} کا تابع ہوگا۔   (سوال  \حوالہ{سوال_حر_وقت_کا_تابع_مقدار_معلوم}  میں  \عددی{R(t)}،  پھیلتے ہوئے چوکور کنویں کی،  چوڑائی  ہوگی۔)  یوں
\begin{align}\label{مساوات_حر_ایک_مقدار_معلوم}
\frac{\partial \psi_n}{\partial t} = \frac{\partial \psi_n}{\partial R} \frac{\dif R}{\dif t}
\end{align}
لہٰذا 
\begin{align}\label{مساوات_حر_ہندسی_ہیّت_فرق}
\gamma_n (t) = i \int_0^t \big\langle \psi_n \big| \frac{\partial \psi_n}{\partial R} \big\rangle \frac{\dif R}{\dif t'} \dif t' = i \int_{R_i}^{R_f} \big\langle \psi_n \big | \frac{\partial \psi_n}{\partial R}\big \rangle \dif R 
\end{align}
ہو گا، جہاں \عددی{R_i} اور \عددی{R_f} مقدار معلوم \عددی{R_t} کی بالترتیب ابتدائی اور اختتامی قیمتیں ہوں گی۔ بالخصوص،  اگر وقت  \عددی{ T} کے  بعد ہیملٹنی واپس اپنا  ابتدائی روپ اختیار کرے تب \عددی{R_f = R_i} لہٰذا \عددی{\gamma_n (T) = 0} ہوگا،  جو زیادہ دلچسپ  صورتحال نہیں!

 میں نے مساوات  \حوالہ{مساوات_حر_ایک_مقدار_معلوم}  میں فرض کیا کہ ہیملٹنی میں صرف ایک مقدار معلوم ایسا ہے جو تبدیل ہوتا ہو۔ اب  فرض کریں \عددی{N} عدد مقدار معلوم \عددی{R_1 (t)}،  \عددی{R_2 (t)}، \عددی{\dotsc}، \عددی{R_N (t)} تبدیل ہوتے ہوں؛  تب درج ذیل ہوگا 
\begin{align}
\frac{\partial \psi_n}{\partial t} = \frac{\partial \psi_n}{\partial R_1} \frac{\dif R_1}{\dif t} + \frac{\partial \psi_n}{\partial R_2} \frac{\dif R_2}{\dif t} + \cdots + \frac{\partial \psi_n}{\partial R_N} \frac{\dif R_N}{\dif t} = (\nabla_R \psi_n) \cdot \frac{\dif \kvec{R}}{\dif t}
\end{align} 
جہاں \عددی{\kvec{R} \equiv (R_1, R_2, \dotsc , R_N)} ہے اور \عددی{\nabla_R} ان مقدار معلوم کے لحاظ سے ڈھلوان ہے۔  اس مرتبہ درج ذیل ہوگا 
\begin{align}
\gamma_n (t) = i \int_{\kvec{R}_i}^{\kvec{R}_f} \langle \psi_n | \nabla_R \psi_n \rangle \cdot \dif \kvec{R}
\end{align}
اور اگر وقت \عددی{ T} کے بعد ہیملٹنی واپس اپنا  اصل روپ اختیار کرتا ہو تب خالص  ہندسی ہیّتی تبدیلی درج ذیل ہوگی۔
\begin{align}\label{مساوات_حر_غیر_صفر_لکیری_تکمل}
\gamma_n (T) = i \oint \langle \psi_n | \nabla_R \psi_n \rangle \cdot \dif \kvec{R}
\end{align}
یہ مقدار معلوم   فضا میں  بند راہ پر  \ترچھا{لکیری}  تکمل ہے، جو عموماً غیر صفر ہوگا۔ مساوات  \حوالہ{مساوات_حر_غیر_صفر_لکیری_تکمل}  کو پہلی مرتبہ  \سن{ 1984}  میں\حاشیہد{حیرت کی بات ہے کہ \عددی{60} سال تک یہ حقیقت کسی کو نظر نہیں آئی۔}   \ترچھا{میکائل   بیری}  نے حاصل کیا اور یوں \عددی{\gamma_n (T)} \اصطلاح{ ہیّت بیری}\فرہنگ{ہیّت بیری}\حاشیہب{Berry's phase}\فرہنگ{Berry's phase}  کہلاتی ہے۔ دھیان رہے  کہ ( جب تک  حرکت  اتنی آہستہ ہو کہ  حر ناگزر کے شرائط مطمئن ہوتے ہوں)  \عددی{\gamma_n (T)} کی قیمت صرف اس راہ پر منحصر ہوگی جس پر چلا جائے نہ کہ راہ پر چلنے کی رفتار پر۔  اس کے برعکس،  مجموعی حرکی ہیّت 
\begin{align*}
\theta_n (T) = - \frac{1}{\hslash} \int_0^T E_n (t') \dif t'
\end{align*}
گزرے ہوئے وقت کے  تابع ہوگی۔

 ہم اس سوچ کے عادی ہیں کہ تفاعل موج کی  ہیّت اختیاری  ہے؛  طبیعی مقداروں میں  \عددی{\abs{\Psi}^2} پایا جاتا ہے، لہٰذا   ہیّتی جزو ضربی  کٹ جاتا ہے۔  اسی لیے عموماً لوگوں کا خیال تھا کہ ہندسی ہیّت کی کوئی طبیعی اہمیت نہیں؛  آخر \عددی{\psi_n (t)} کی  ہیّت بھی تو  اختیاری ہے۔  یہ  بیری کی دور اندیشی تھی  کہ انہوں نے اس حقیقت کو پہچانا کہ ہیملٹنی کو  بند دائرے پر  پھیرا دے کر  واپس اپنے  اصل روپ میں لانے سے ابتدا اور اختتام کے بیچ  زائد     ہیّت  غیر اختیاری ہوگی،    جس کی پیمائش بھی  کی جا سکتی  ہے۔
 
  مثال کے طور پر،  ذرات ( تمام حال \عددی{\Psi} میں)  کی ایک شعاع کو دو حصوں میں تقسیم کرکے،  صرف ایک حصے کو حر ناگزر تبدیل ہوتے مخفیہ سے گزارا جاتا ہے۔ دونوں حصوں کو دوبارہ اکٹھا کرنے سے درج ذیل روپ کا  مجموعی تفاعل موج  حاصل ہوگا 
\begin{align}
\Psi = \frac{1}{2} \Psi_0 + \frac{1}{2} \Psi_0 e^{i \Gamma}
\end{align}
جہاں  \عددی{\Psi_0} \قول{ سیدھی پہنچتی}   شعاع کا تفاعل موج   اور  \عددی{\Gamma}   تغیر پذیر  \عددی{H} کی بنا پر شعاع کی   \ترچھا{زائد}   ہیّت ہے  (جس کا کچھ حصہ حرکی اور کچھ ہندسی ہوگا)۔   اس صورت میں درج ذیل ہوگا۔
\begin{align}\label{مساوات_حر_ہیّتی_انتقال}
\abs{\Psi}^2 &= \frac{1}{4} \abs{\Psi_0}^2 \big ( 1 + e^{i \Gamma} \big ) \big ( 1 + e^{- i \Gamma} \big ) \nonumber\\
&= \frac{1}{2} \abs{\Psi_0}^2 (1 + \cos \Gamma) = \abs{\Psi_0}^2 \cos^2 (\Gamma/2)
\end{align}
یوں  تعمیری  اور تباہ کن\اصطلاح{ مداخلت}\فرہنگ{مداخلت}\حاشیہب{interference}\فرہنگ{interference} کے نقاط   ( جہاں \عددی{\Gamma} کی قیمت \عددی{\pi} کی بالترتیب جفت اور طاق مضرب ہوگی)  سے   \عددی{\Gamma} کی پیمائش کی جا سکتی ہے ( بیری اور دیگر مصنفین کو شبہ تھا کہ زیادہ بڑی حرکی ہیّت کی موجودگی میں ہندسی  ہیّت  نظر نہیں آئے گی،  لیکن انہیں علیحدہ علیحدہ  کرنا ممکن ثابت ہوا ہے)۔ 

 تین ابعادی مقدار معلوم فضا،  \عددی{\kvec{R} = (R_1, R_2, R_3)}،    میں کلیہ  بیری  (مساوات \حوالہ{مساوات_حر_غیر_صفر_لکیری_تکمل})  سمتی مخفیہ \عددی{\kvec{A}} کی صورت میں \اصطلاح{مقناطیسی بہاو}\فرہنگ{مقناطیسی بہاو}\حاشیہب{magnetic flux}\فرہنگ{magnetic flux}   کے  کلیہ کا  یاد دلاتا ہے۔ سطح \عددی{S} جس کی سرحد منحنی \عددی{ C} ہو سے درج ذیل  بہاو گزرتا ہے  (شکل \حوالہ{شکل_حرارت_نا_گزر_مقناطیسی_بہاو})۔ 
\begin{align}
\Phi \equiv \int_S \kvec{B} \cdot \dif \kvec{a}
\end{align}

مقناطیسی میدان کو سمتی مخفیہ کے  روپ \عددی{(\kvec{B} = \nabla \times \kvec{A})}  میں لکھ کر مسئلہ سٹوکس کے  اطلاق سے درج ذیل حاصل ہوگا۔
\begin{align}
\Phi = \int_S (\nabla \times \kvec{A}) \cdot \dif \kvec{a} = \oint_C \kvec{A} \cdot \dif \kvec{r}
\end{align}
یوں ہیّت بیری کو  مقدار معلوم فضا میں بند راہ کے اندر سے \قول{مقناطیسی میدان}   کا  \قول{بہاو}
\begin{align}\label{مساوات_حر_مماثلت}
\kvec{``B"} = i \nabla_R \times \langle \psi_n | \nabla_R \psi_n \rangle
\end{align}
تصور کیا جا سکتا ہے۔  اس کو دوسری طرف سے بھی بیان کیا جا سکتا ہے:  تین ابعادی صورت  میں ہیّت بیری کو  سطحی تکمل:
\begin{align}\label{مساوات_حر_سطحی_تکمل}
\gamma_n (T) = i \int [\nabla_R \times \langle \psi_n | \nabla_R \psi_n \rangle ] \cdot \dif \kvec{a}
\end{align}
  لکھا جا سکتا ہے۔ اس مقناطیسی مماثلت کو کافی دور تک لے جایا جا سکتا ہے،  تاہم ہماری  مقاصد  کے نقطہ نظر سے مساوات  \حوالہ{مساوات_حر_سطحی_تکمل}  محض \عددی{\gamma_n (T)}  لکھنے کا دوسرا انداز ہے۔

ہیّت بیری کی کلاسیکی مثال  مستقل مقدار کے  مقناطیسی میدان،  جس کی سمت تبدیل ہوتی ہو،  میں مبدا پر  الیکٹران ہے۔  پہلے اس مخصوص صورت (جس کا تجزیہ مثال  \حوالہ{مثال_حر_مخصوص_صورت}   میں کیا گیا)   پر غور کرتے  ہیں   جس میں محور \عددی{z} کے ساتھ   مقررہ  زاویہ \عددی{\alpha}  پر رہتے ہوئے،   مستقل زاویائی  سمتی رفتار \عددی{\omega} سے،    \عددی{\kvec{B} (t)}  استقبالی حرکت کرتا ہے۔ (میدان \عددی{B}  کی ہم راہ \قول{  ہم میدان}  الیکٹران کے لئے)  مساوات \حوالہ{مساوات_حر_حل_خطی_جوڑ} ٹھیک ٹھیک حل دیتی ہے۔ حر ناگزر  طریق،  \عددی{\omega \ll \omega_1}،  میں 
\begin{align}
\lambda = \omega_1 \sqrt{1 - 2 \frac{\omega}{\omega_1} \cos \alpha + \big ( \frac{w}{w_1} \big )^2 } \cong \omega_1 \big ( 1 - \frac{\omega}{\omega_1} \cos \alpha \big ) = \omega_1 - \omega \cos \alpha
\end{align}
ہوگا،  لہٰذا مساوات \حوالہ{مساوات_حر_حل_خطی_جوڑ} درج ذیل روپ اختیار کرتی ہے۔ 
\begin{align}
\chi (t) \cong \,&e^{- i \omega_1 t/2} e^{i (\omega \cos \alpha) t/2} e^{- i \omega t/2} \chi_+(t) \nonumber\\
&+i \big [ \frac{\omega}{\omega_1} \sin \alpha \sin \big ( \frac{\omega_1 t}{2} \big ) \big ]e^{+ i \omega t/2} \chi_{-} (t)
\end{align}
دوسرے جزو کو  \عددی{\omega/\omega_1 \to 0} کی صورت میں رد کرتے ہوئے حر ناگزر روپ کے مطابق  نتیجہ حاصل ہو گا (مساوات  \حوالہ{مساوات_حر_وہیں_رہتا_ہے})۔ حرکی ہیّت درج ذیل ہے 
\begin{align}
\theta_+ (t) = - \frac{1}{\hslash} \int_0^t E_+ (t') \dif t' = - \frac{\omega_1 t}{2}
\end{align}
(جہاں مساوات \حوالہ{مساوات_حر_حرکی_ہیّت_توانائی} سے \عددی{E_{+} = \hslash \omega_1 /2} ہوگا)،  لہٰذا ہندسی ہیّت درج ذیل ہوگی۔
\begin{align}
\gamma_+ (t) = (\cos \alpha - 1) \frac{\omega t}{2}
\end{align}
ایک مکمل پھیرے کے لیے \عددی{T = 2\pi / \omega} ہوگا،  لہٰذا ہیّت بیری درج ذیل ہوگی۔
\begin{align}\label{مساوات_حر_گاما}
\gamma_+ (T) = \pi (\cos \alpha - 1)
\end{align}

اب  زیادہ عمومی صورت پر غور کرتے ہیں،  جس میں مقناطیسی میدان سمتیہ کی  نوک رداس \عددی{r = B_0}  کرہ  کی  سطح پر  \ترچھا{ اختیاری}  بند راہ جھاڑتی ہے  (شکل  \حوالہ{شکل_حرارت_نا_گزر_مقناطیسی_میدان_بند_راہ})۔ میدان \عددی{\kvec{B} (t)} کی ہم راہ  ہم میدان کو ظاہر کرنے والا امتیازی حال درج ذیل روپ کا ہوگا (سوال \حوالہ{سوال_تین_بعدی_قالب_ایس_زاویائی_معیار_حرکت} دیکھیں)
\begin{align}
\chi_{+} = 
\begin{pmatrix}
\cos (\theta/2) \\[0.5em]
e^{i \phi} \sin (\theta /2)
\end{pmatrix}
\end{align}
جہاں \عددی{\kvec{B}} کے  کروی محدد \عددی{\theta} اور \عددی{\pi}   اب وقت کے تفاعلات ہیں۔ کروی محدد میں ڈھلوان  درج ذیل ہوگا،  جیسے آپ جدول سے دیکھ سکتے ہیں۔ 
\begin{align}
\nabla \chi_{+} &= \frac{\partial \chi_{+}}{\partial r} \ar + \frac{1}{r} \frac{\partial \chi_{+}}{\partial \theta} \atheta + \frac{1}{r \sin \theta} \frac{\partial \chi_{+}}{\partial \phi} \aphi \nonumber\\
&= \frac{1}{r} 
\begin{pmatrix}
- (1/2) \sin (\theta/2) \\[0.5em]
(1/2) e^{i \phi} \cos (\theta /2)
\end{pmatrix}
\atheta + \frac{1}{r \sin \theta}
\begin{pmatrix}
0 \\[0.5em]
i e^{i \phi} \sin (\theta /2)
\end{pmatrix} 
\aphi
\end{align}
یوں درج ذیل ہوگا۔
\begin{align}
\langle \chi_{+} | \nabla \chi_{+} \rangle &= \frac{1}{2r} \big [ - \sin \frac{\theta}{2} \cos \frac{\theta}{2} \atheta + \sin \frac{\theta}{2} \cos \frac{\theta}{2} \atheta + 2i \,\frac{\sin^2 (\theta/2)}{\sin \theta} \aphi \big ] \nonumber\\
&= i \,\frac{\sin^2 (\theta /2)}{r \sin \theta}\aphi
\end{align}
مساوات \حوالہ{مساوات_حر_سطحی_تکمل} کے لیے ہمیں اس  مقدار کی گردش درکار ہوگی۔
\begin{align}
\nabla \times \langle \chi_{+} | \nabla \chi_{+} \rangle = \frac{1}{r \sin \theta} \frac{\partial}{\partial \theta} \big [ \sin \theta \big ( \frac{i \sin^2 (\theta /2)}{r \sin \theta} \big ) \big ] \ar = \frac{i}{2r^2} \ar
\end{align}
یوں مساوات \حوالہ{مساوات_حر_سطحی_تکمل} کے تحت درج ذیل ہوگا۔
\begin{align}
\gamma_{+} (T) = - \frac{1}{2} \int \frac{1}{r^2} \ar \cdot \dif \kvec{a}
\end{align}
تکمل کرہ کی سطح پر اس رقبے پر لیا جائے گا جس کو \عددی{\kvec{B}} کی  نوک  ایک پھیرے  میں  جھاڑتی  ہے،   لہٰذا \عددی{\dif \kvec{a} = r^2 \dif \Omega \ar}  اور 
\begin{align}\label{مساوات_حر_عمومی_نتیجہ}
\gamma_{+} (T) = - \frac{1}{2} \int \dif \Omega = - \frac{1}{2} \Omega
\end{align}
ہوگا، جہاں مبدا  پر ٹھوس زاویہ \عددی{\Omega} ہے۔  یہ ایک انتہائی سادہ نتیجہ ہے،  جو ہمیں اس کلاسیکی مسئلے  کا یاد دلاتا ہے جس سے ہم نے یہ تبصرہ شروع کیا  (یعنی زمین کی سطح پر  بند راہ پر   بلا رگڑ   رقاص   کی منتقلی)۔  اس نتیجے کے تحت،  کسی اختیاری بند راہ پر،   مقناطیس کی مدد سے الیکٹران کے چکر کو حر ناگزر  پھیرا دینے سے،  خالص ( ہندسی) تبدیلی ہیّت مقناطیسی میدان سمتیہ کے جھاڑنے کے  ٹھوس زاویہ کی منفی آدھی   ہوگی۔  مساوات  \حوالہ{مساوات_حر_ٹھوس_زاویہ}  کو مد نظر رکھتے ہوئے یہ عمومی نتیجہ   مخصوص نتیجہ   (مساوات \حوالہ{مساوات_حر_گاما})   کے مطابق ہے،  جیسا یقیناً ہونا بھی چاہیے  تھا۔
\انتہا{مثال}
\ابتدا{سوال}

 \جزوحصہ{اہارونو و بوہم اثر}\شناخت{حصہ_حرارت_نا_گزر_اہارونوبوہم_اثر}
کلاسیکی برقی حرکیات میں؛ مخفیے  ( \عددی{\varphi} اور \عددی{\kvec{A}}) \حاشیہد{کوانٹائی میکانیات میں روایتی طور پر  حرف \عددی{V} کو مخفی تونائی کے لئے استعمال کیا جاتا ہے،لیکن  برقی حرکیات میں یہی حرف غیر سمتی مخفیہ کے لئے استعمال کیا جاتا ہے۔ غلط فہمی سے بچنے کے لئے غیر سمتی مخفیہ کے لئے حرف \عددی{\varphi} استعمال کروں گا۔ اس حصہ کے پس منظر کے لئے سوال \حوالہ{سوال_تین_ابعادی_کلیہ_قوت_لورنز}، سوال \حوالہ{سوال_تین_ابعادی_کلیہ_قوت_لورنز_بلا} اور سوال \حوالہ{سوال_تین_ابعادی_کلیہ_قوت_لورنز_بلا_بلا} دیکھیں۔} بلاواسطہ ناقابل پیمائش ہیں؛   برقی اور مقناطیسی  \ترچھا{میدان}: 
\begin{align}\label{مساوات_حر_طبیعی_میدان}
\kvec{E} = - \nabla \varphi - \frac{\partial \kvec{A}}{\partial t}, \quad \kvec{B} = \nabla \times \kvec{A}
\end{align}
\ترچھا{ طبیعی}  مقادیر ہیں۔   بنیادی قوانین ( میکسویل مساوات اور لورنز  قوت  قاعدہ) مخفیوں  کا کوئی ذکر نہیں کرتے،  جو  (منطقی نقطہ نظر سے)  نظریہ  مرتب کرنے  کے لیے کار آمد لیکن ویسے غیر ضروری ہیں۔  یقیناً، آپ  بغیر  خوف  و خطر ان مخفیوں  کو تبدیل کر سکتے ہیں: 
\begin{align}
\varphi \to \varphi' = \varphi - \frac{\partial \Lambda}{\partial t}, \quad \kvec{A} \to \kvec{A}' = \kvec{A} + \nabla \Lambda
\end{align}
جہاں \عددی{\Lambda} مقام اور وقت کا کوئی بھی تفاعل ہو سکتا ہے؛  یہ  \اصطلاح{ماپ تبادلہ}\فرہنگ{ماپ تبادلہ}\حاشیہب{gauge transformation}\فرہنگ{gauge transformation}  کہلاتا  ہے،   جس کا میدانوں پر کوئی اثر نہیں    ( جیسا آپ مساوات  \حوالہ{مساوات_حر_طبیعی_میدان}  استعمال کرتے ہوئے   دیکھ سکتے ہیں)۔ 

 کوانٹائی میکانیات میں مخفیے  زیادہ اہم کردار ادا کرتے ہیں،  چونکہ ہیملٹنی کو \عددی{\varphi} اور \عددی{\kvec{A}} کی صورت میں بیان کیا جاتا ہے:
\begin{align}\label{مساوات_حر_ہیملٹنی_اہارونو}
H = \frac{1}{2m} \big ( \frac{\hslash}{i} \nabla - q \kvec{A} \big )^2 + q \varphi
\end{align}
  نہ کہ \عددی{\kvec{E}} اور \عددی{\kvec{B}} کی صورت میں۔  بہرحال،   زیر ماپ تبادلہ  کوانٹائی   نظریہ  غیر متغیر رہتا  ہے  (سوال   \حوالہ{سوال_تین_ابعادی_کلیہ_قوت_لورنز_بلا_بلا}  دیکھیں)،   اور بہت لمبے  عرصے  کے لیے مانا جاتا تھا  کہ  جن خطوں میں  \عددی{\kvec{E}} اور \عددی{\kvec{B}} صفر ہوں وہاں، بالکل  کلاسیکی نظریہ کی طرح،  کسی قسم کا برقناطیسی اثر نہیں پایا جاتا۔  لیکن \سن {1959} میں اہارونو اور بوہم نے دکھایا کہ اس خطہ میں بھی،  جہاں میدان صفر ہو، حرکت پذیر باردار ذرہ کے کوانٹائی رویہ  پر   سمتی مخفیہ  اثر انداز ہوگا۔  میں ایک سادہ مثال پیش کر کے  اہارونو و بوہم   اثر پر تبصرہ کروں گا اور اس کے بعد اس کا  ہیّت بیری کے ساتھ تعلق دکھاوں  گا۔

 فرض کریں ایک ذرے  کو رداس \عددی{b} کے دائرے  پر رہنے کا پابند بنایا جاتا ہے۔  اس دائرے کے محور پر رداس \عددی{a < b} کا  لمبا    \اصطلاح{پیچواں  لچھا}\فرہنگ{پیچواں لچھا}\حاشیہب{solenoid}\فرہنگ{solenoid}     ہے جس میں یک سمتی برقی رو \عددی{I} ہے  (شکل  \حوالہ{شکل_حرارت_نا_گزر_پیچواں_دائرہ_حرکت})۔ بہت لمبے   پیچواں لچھے  کی صورت میں،   پیچواں لچھے کے اندر مقناطیسی میدان یکساں ہو گا،    جبکہ  اس  کے باہر  میدان صفر ہوگا۔  تاہم پیچواں لچھے کے باہر  سمتی مخفیہ غیر صفر ہوگا؛   یقیناً ( \عددی{\nabla \cdot \kvec{A} = 0} کی  موزوں ماپ شرط لیتے ہوئے)     درج ذیل ہوگا 
\begin{align}\label{مساوات_حر_سمتی_مخفیہ}
\kvec{A}& = \frac{\Phi}{2\pi r} \aphi &&  (r > a)
\end{align}
جہاں \عددی{\Phi = \pi a^2 B} پیچواں لچھے سے گزرتا ہوا \اصطلاح{  مقناطیسی بہاو}\فرہنگ{مقناطیسی بہاو}\حاشیہب{magnetic flux}\فرہنگ{magnetic flux}  ہوگا۔  پیچواں لچھا خود غیر باردار ہے،  لہٰذا غیر سمتی مخفیہ \عددی{\varphi} صفر ہو گا۔  ایسی صورت میں ہیملٹنی ( مساوات   \حوالہ{مساوات_حر_ہیملٹنی_اہارونو})   درج ذیل روپ اختیار کرتی ہے۔ 
\begin{align}\label{مساوات_حر_ہیملٹنی_اہارونو_حاصل}
H = \frac{1}{2m} [- \hslash^2 \nabla^2 + q^2 A^2 + 2 i \hslash q \kvec{A} \cdot \nabla]
\end{align}
 تفاعل موج صرف  اسّمتی زاویہ \عددی{\phi}   \عددی{(\theta=\pi/2, r=b) } کا تابع  ہے،  لہٰذا \عددی{\nabla \to (\aphi /b)(\dif / \dif \phi)} ہوگا،  اور مساوات شروڈنگر درج ذیل لکھی جائے گی۔
\begin{align}
\frac{1}{2m} \big [ - \frac{\hslash^2}{b^2} \frac{\dif^{\,2}}{\dif \phi^{\,2}} + \big ( \frac{q \Phi}{2 \pi b} \big )^2 + i \frac{h q \Phi}{\pi b^2} \frac{\dif}{\dif \phi} \big ] \psi (\phi) = E \psi (\phi)
\end{align}
یہ مستقل عددی سروں والی خطی تفرقی مساوات ہے:
\begin{align}
\frac{\dif^{\,2} \psi}{\dif \phi^{\,2}} - 2i \beta \frac{\dif \psi}{\dif \phi} + \epsilon \psi = 0
\end{align}
جہاں درج ذیل ہیں۔
\begin{align}
\beta \equiv \frac{q \Phi}{2 \pi \hslash} \quad \text{\RL{اور}}\quad  \epsilon \equiv \frac{2 m b ^2 E}{\hslash^2} - \beta^2
\end{align}
اس کے حل درج ذیل روپ کے ہونگے 
\begin{align}
\psi = Ae^{i \lambda \phi}
\end{align}
جہاں درج ذیل ہوگا۔
\begin{align}
\lambda = \beta \pm \sqrt{ \beta^2 + \epsilon} = \beta \pm \frac{b}{\hslash} \sqrt{2 m E}
\end{align}
نقطہ \عددی{\phi = 2\pi} پر \عددی{\psi (\phi)} کے  استمرار کی بنا پر \عددی{\lambda} عدد صحیح ہو گا:
\begin{align}
\beta \pm \frac{b}{\hslash} \sqrt{2m E} = n
\end{align} 
 جس سے درج ذیل حاصل  ہوتا ہے۔
\begin{align} 
E_n = \frac{\hslash^2}{2m b^2} \big ( n - \frac{q \Phi}{2 \pi \hslash} \big )^2, \quad (n = 0, \pm 1, \pm 2, \dotsc)
\end{align}

پیچواں لچھا   دائرے پر ذرے  کا  دو پٹرتا  انحطاط ختم کرتا ہے  (سوال  \حوالہ{سوال_غیر_تابع_شروڈنگر_بے_رگڑ_موتی}):   مثبت \عددی{n} ، جو پیچواں لچھے میں رو کے رخ حرکت کرتے ہوئے ذرے  کو ظاہر کرتا ہے  (\عددی{ q} مثبت فرض کرتے  ہوئے) ، کی توانائی  منفی \عددی{n} کے لحاظ سے،  جو مخالف رخ ذرے  کو ظاہر کرتا ہے،   کم ہو گی۔  زیادہ اہم بات یہ ہے کہ،  اجازتی توانائیوں کا دارومدار پیچواں لچھے کے اندر میدان پر ہوگا،  اگرچہ اس مقام پر جہاں ذرہ پایا جاتا ہے میدان صفر ہے!\حاشیہد{\اصطلاح{اعلٰی موصلی}\فرہنگ{اعلٰی موصلی}\تحریر{(superconducting)}\فرہنگ{superconducting}  چھلوں کی ایک مخصوص خاصیت یہ ہے کہ ان کے احاطہ میں بہاو کوانٹائی ہوتا ہے: \عددی{\Phi=(2\pi \hslash/q)n'}، جہاں \عددی{n'} عدد صحیح ہو گا۔ایسی صورت میں یہ اثر قابل  کشف نہیں ہو گا، چونکہ \عددی{E_n=(\hslash^2/2mb^2)(n+n')^2} ہو گا اور \عددی{(n+n')} بھی  عدد صحیح ہے۔ اتفاق کی بات ہے کہ یہاں بار \عددی{q} الیکٹران کے بار کا دگنا ہو گا؛ (اعلٰی موصلی الیکٹران جوڑوں  کی صورت  میں رہتے ہیں۔) لیکن  \اصطلاح{کوانٹازنی بہاو}\فرہنگ{کوانٹازنی بہاو}\تحریر{(flux quantization)}\فرہنگ{flux quantization}  کو اعلٰی موصل   نافذ  کرتا ہے (جو دائری برقی رو کے امالہ کے ذریعہ فرق کو ختم کرتا ہے)، نہ کہ پیچواں لچھا یا برقناطیسی میدان، اور یہ (غیر اعلٰی موصل) مثال میں نہیں پایا جاتا، جس پر یہاں غور کیا گیا ہے۔}

 زیادہ عمومی صورت  پر غور کرنے کی خاطر،  فرض کریں ایک ذرہ  ایسے خطے  میں حرکت کرتا ہے جہاں \عددی{\kvec{B}}  صفر ہے (لہٰذا \عددی{\nabla \times \kvec{A} = 0} ہوگا )، لیکن  \عددی{\kvec{A}} خود غیر صفر ہو گا۔ ( اگرچہ میں فرض کرتا ہوں کہ \عددی{\kvec{A}} ساکن ہے،  اس ترکیب کو تابع وقت مخفیے  کے لئے عمومیت دی جا سکتی ہے۔) مخفی توانائی \عددی{V}،  جس میں برقی حصہ \عددی{q\varphi}  شامل یا غیر شامل ہو سکتا ہے،  کی (تابع وقت)   مساوات شروڈنگر 
\begin{align}\label{مساوات_حر_تابع_وقت_مساوات_شروڈنگر_اہارونو}
\big [ \frac{1}{2m} \big ( \frac{\hslash}{i} \nabla - q \kvec{A} \big )^2 + V \big ] \Psi = i \hslash \frac{\partial \Psi}{\partial t}
\end{align}
کی سادہ روپ درج ذیل لکھ کر حاصل کی جا سکتی ہے 
\begin{align}
\Psi = e^{i g} \Psi'
\end{align}
جہاں  
\begin{align}
g (\kvec{r}) \equiv \frac{q}{\hslash} \int_{\mathcal{O}}^{\kvec{r}} \kvec{A} (\kvec{r}') \cdot \dif \kvec{r}'
\end{align}
ہے اور \عددی{\mathcal{O}} کوئی  (اختیاری منتخب)  نقطہ حوالہ ہے۔  دھیان رہے کہ یہ تعریف صرف اس صورت با معنی ہوگی جب پورے  خطے میں \عددی{\nabla \times \kvec{A} = 0} ہو؛  ورنہ لکیری تکمل \عددی{\mathcal{O}} سے \عددی{\kvec{r}} تک راہ پر منحصر ہوگا،  اور یوں \عددی{\kvec{r}} کا تفاعل نہیں ہوگا۔  \عددی{\Psi'} کی صورت میں \عددی{\Psi} کی ڈھلوان  
\begin{align*}
\nabla \Psi = e^{i g} (i \nabla g) \Psi' + e^{ig}(\nabla \Psi')
\end{align*}
ہو گی، لیکن \عددی{\nabla g = (q/ \hslash) \kvec{A}} ہے،  لہٰذا 
\begin{align}\label{مساوات_حر_ہیملٹنی_اہارونو_حاصل_الف}
\big ( \frac{\hslash}{i} \nabla - q \kvec{A} \big ) \Psi = \frac{\hslash}{i} e^{i g} \nabla \Psi'
\end{align}
اور یوں درج ذیل ہوگا۔
\begin{align}\label{مساوات_حر_ہیملٹنی_اہارونو_حاصل_ب}
\big ( \frac{\hslash}{i} \nabla - q \kvec{A} \big )^2 \Psi = - \hslash^2 e^{i g} \nabla^2 \Psi'
\end{align}
اس کو مساوات  \حوالہ{مساوات_حر_تابع_وقت_مساوات_شروڈنگر_اہارونو}  میں پُر کر کے مشترک جزو ضربی \عددی{e^{i g}} کو کاٹ کر درج ذیل ملتا ہے۔
\begin{align}\label{مساوات_حر_مساوات_شروڈنگر_بغیر_اے}
- \frac{\hslash^2}{2m} \nabla^2 \Psi' + V \Psi' = i \hslash \frac{\partial \Psi'}{\partial t}
\end{align}

بظاہر\ترچھا{ بغیر}  \عددی{\kvec{A}} مساوات شروڈنگر کو  \عددی{\Psi'}  مطمئن کرتا ہے۔  مساوات  \حوالہ{مساوات_حر_مساوات_شروڈنگر_بغیر_اے}  کا حل تلاش کرنے کے بعد  (بغیر گردش)  سمتی مخفیہ  کے شمول کی تصحیح    حقیر سا کام ہے:  صرف ہیّتی جزو ضربی \عددی{e^{i g}} ساتھ منسلک کرنا ہوگا۔

 اہارونو  اور بوہم نے ایک تجربہ تجویز کیا، جس میں الیکٹران کی شعاع کو دو حصوں میں تقسیم کر کے لمبے پیچواں لچھے کے دونوں اطراف سے گزار کر دوبارہ اکٹھا کیا جاتا ہے  (شکل   \حوالہ{شکل_حرارت_نا_گزر_اہارانووبوہم})۔   ان شعاعوں کو  پیچواں لچھے  سے اتنا دور رکھا جاتا ہے کہ شعاع صرف  ان مقامات سے گزرتی ہے  جہاں \عددی{\kvec{B} = 0} ہو تا ہے۔تاہم \عددی{\kvec{A}}،  جسے مساوات  \حوالہ{مساوات_حر_سمتی_مخفیہ}  پیش کرتی ہے، غیر صفر ہوگا،  اور (دونوں اطراف \عددی{V} کی قیمت ایک  جیسی تصور کرتے ہوئے)  اختتامی  نقطے  پر دونوں شعاعوں  کی  ہیّت:
\begin{align}
g = \frac{q}{\hslash} \int \kvec{A} \cdot \dif \kvec{r} = \frac{q \Phi}{2 \pi \hslash} \int \big ( \frac{1}{r} \aphi \big ) \cdot (r \aphi \dif \phi ) = \pm \frac{q \Phi}{2 \hslash}
\end{align}
  مختلف ہو گی۔  یہاں مثبت علامت ان الیکٹران کے لیے ہے  جو   \عددی{\kvec{A}} کے رخ حرکت کرتے ہیں؛ یعنی پیچواں لچھے میں برقی رو کے رخ۔  دونوں شعاعوں کے بیچ ہیّتی فرق اس مقناطیسی بہاو کے راست متناسب ہوگا جسے ان کی راہ گھیرتی ہیں۔ 
\begin{align}\label{مساوات_حر_اہارونو_بوہم_نتیجہ}
\text{\RL{ہیّتی فرق}}= \frac{q \Phi}{\hslash}
\end{align}
اس ہیّتی  انتقال  سے قابل پیمائش مداخلت  (مساوات  \حوالہ{مساوات_حر_ہیّتی_انتقال})   پیدا ہوتی ہے جس کی تجرباتی تصدیق چیمبرز اور ساتھی کر چکے ہیں۔

 اہارونو و بوہم اثر کو ہندسی ہیّت کی ایک مثال تصور کی جا سکتی ہے۔  فرض کریں مخفیہ \عددی{ V(\kvec{r} - \kvec{R})}  بار دار ذرے  کو ایک ڈبے  میں رہنے کا پابند بناتا ہے (ڈبے کا مرکز  پیچواں لچھے سے باہر نقطہ \عددی{\kvec{R}} پر ہے)؛ شکل \حوالہ{شکل_حرارت_نا_گزر_مقید_ذرہ_ڈبیہ} دیکھیں۔(     ہم کچھ ہی دیر میں اس ڈبے  کو  پیچواں لچھے کے گرد ایک پھیرا دیں گے،  لہٰذا \عددی{\kvec{R}} وقت کا تفاعل ہوگا،  تاہم ابھی اسے ایک غیر متغیر سمتیہ تصور کریں۔) اس ہیملٹنی کے امتیازی تفاعلات کا تعین   درج ذیل کرتی ہے۔

\end{align}

\end{align}
لیتے ہے جہاں\حاشیہد{حوالہ نقطے \عددی{\mathcal{O}} کو ڈبے کے وسط پر منتخب کرنا سود مند ثابت ہوتا ہے، چونکہ ایسا کرنا ضمانت دیتا ہے کہ  پیچواں لچھے کے گرد ایک پھیرا مکمل کرنے سے  ابتدائی  ہیّتی  روایت  حاصل ہو گی۔ اگر آپ، مثلاً،  مقررہ  فضا میں حوالہ نقطہ منتخب کریں، تب راہ پیچواں لچھے کے گرد لپٹی ہو گی، اور ایسے خطے کو گھیرے گی جہاں \عددی{\kvec{A}} کی گردش غیر صفر ہے، لہٰذا   آپ کو  آخری نقطے پر \قول{ہاتھ سے}  ہیّت درست کرنی ہو گی۔ اگرچہ، اب بھی وہی جواب حاصل ہو گا، تاہم جواب حاصل کرنے کا یہ بہتر طریقہ نہیں ہے۔عموماً، مساوات \حوالہ{مساوات_حر_تابع_وقت_ہیملٹنی_توانائی} میں  تفاعلات موج کی ہیّتی  روایت طے  کرتے ہوئے، ہم چاہیں گے کہ \عددی{\psi_n(x,T)=\psi_n(x,0)} ہو تا کہ  جعلی  ہیّتی فرق شامل نہ ہوں۔}
\begin{align}
g \equiv \frac{q}{\hslash} \int_{\kvec{R}}^{\kvec{r}} \kvec{A} (\kvec{r}') \cdot \dif (\kvec{r}')
\end{align}
ہے، اور  \عددی{\kvec{A} \to 0} کی صورت میں  \عددی{\psi'} اسی امتیازی قدر مساوات کو  مطمئن کرے گا۔
\begin{align}
\big [ - \frac{\hslash^2}{2m} \nabla^2 + V (\kvec{r} - \kvec{R}) \big ] \psi_n' = E_n \psi'_n
\end{align}
آپ نے دیکھا کہ \عددی{\psi'_n}صرف  ہٹاو  \عددی{(\kvec{r} - \kvec{R})} کا تفاعل ہے،  نہ کہ( \عددی{\psi_n} کی طرح)   \عددی{\kvec{r}} اور \عددی{\kvec{R}} کا  علیحدہ علیحدہ تفاعل۔

 آئیے اب اس ڈبے کو  پیچواں لچھے کے گرد ایک پھیرا دیتے ہیں  ( اس عمل  کو حر ناگزر ہونے کی  بھی ضرورت نہیں)۔ ہیّت بیری کا  تعین کرنے کی خاطر ہمیں
  مقدار \عددی{\langle \psi_n | \nabla \psi_n \rangle} کی قیمت درکار ہوگی۔چونکہ 
\begin{align*}
\nabla_R \psi_n = \nabla_R [e^{i g} \psi'_n (\kvec{r} - \kvec{R})] = -i \frac{q}{\hslash} \kvec{A} (\kvec{R}) e^{i g} \psi'_n (\kvec{r} - \kvec{R}) + e^{i g} \nabla_R \psi'_n (\kvec{r} - \kvec{R})
\end{align*}
ہے لہٰذا درج ذیل  ہو گا۔ 
\begin{align}
\langle \psi_n | \nabla_R \psi_n &\rangle \nonumber\\
&= \int e^{- i g} [\psi'_n (\kvec{r} - \kvec{R})]^{*} e^{i g} \big [ - i \frac{q}{\hslash} \kvec{A} (\kvec{R}) \psi'_n (\kvec{r} - \kvec{R}) + \nabla_R \psi'_n (\kvec{r} - \kvec{R}) \big ] \dif^{\,3} \kvec{r} \nonumber\\
&= - i \frac{q}{\hslash} \kvec{A} (\kvec{R}) - \int [\psi'_n (\kvec{r} - \kvec{R})]^{*} \nabla \psi'_n (\kvec{r} - \kvec{R}) \dif^{\,3} \kvec{r}
\end{align}
بغیر زیرنوشت \عددی{\nabla} متغیر  \عددی{\kvec{r}} کے لحاظ سے ڈھلوان ظاہر کرتا ہے، اور میں  \عددی{(\kvec{r} - \kvec{R})} کے تفاعل پر عمل کے
 دوران \عددی{\nabla_R = - \nabla} بروئے کار لایا۔ لیکن  آخری تکمل کی قیمت  ہیملٹنی \عددی{- (\hslash^2/2m) \nabla^2 + V} کے امتیازی حال میں معیار حرکت کی توقعاتی قیمت
  ضرب  \عددی{i/\hslash} ہے،  جو ہم حصہ \حوالہ{حصہ_غیر_تابع_شروڈنگر_ساکن_حالات}  سے جانتے ہیں کہ \ترچھا{صفر} ہوگا۔ یوں درج ذیل ہوگا۔
\begin{align}
\langle \psi_n | \nabla_R \psi_n \rangle = - i \frac{q}{\hslash} \kvec{A} (\kvec{R})
\end{align}
اس کو کلیہ بیری  (مساوات  \حوالہ{مساوات_حر_غیر_صفر_لکیری_تکمل})    میں پُر کرتے ہوئے درج ذیل نتیجہ  اخذ ہوگا 
\begin{align}
\gamma_n (T) = \frac{q}{\hslash} \oint \kvec{A} (\kvec{R}) \cdot \dif \kvec{R} = \frac{q}{\hslash} \int (\nabla \times \kvec{A}) \cdot \dif \kvec{a} = \frac{q \Phi}{\hslash}
\end{align}
جو اہارونو و بوہم نتیجے   (مساوات  \حوالہ{مساوات_حر_اہارونو_بوہم_نتیجہ})   کی تصدیق کرتا ہے،  اور   واضح  کرتا ہے کہ  اہارونو و بوہم اثر ہندسی ہیّت کی ایک خصوصی صورت ہے۔\حاشیہد{اتفاقاً، موجودہ صورت میں ہیّت بیری اور مقناطیسی بہاو (مساوات \حوالہ{مساوات_حر_مماثلت}) کی تمثیل  تقریباً  مماثل:\عددی{
\kvec{``B"}=\tfrac{q}{\hslash}\kvec{B}
} ہے۔}

اہارونو و بوہم اثر سے ہم کیا مطلب  لیا جائے؟  ظاہر ہے کہ  ہمارا کلاسیکی شعور \ترچھا{ درست نہیں}:    برقناطیسی اثرات ان  خطوں میں پائے جا سکتے ہیں جہاں میدان صفر ہو۔البتہ، یاد  رہے کہ اس سے \عددی{\kvec{A}} خود قابل پیمائش نہیں ہو جاتا؛  صرف  محیط  بہاو   اختتامی  نتیجہ میں پایا جاتا ہے،  اور نظریہ   ماپ  غیر متغیر رہتا ہے۔

ایک ذرہ     (وقفہ \عددی{0 \le x \le a} پر) لامتناہی چوکور کنویں کے زمینی حال سے آغاز کرتا ہے۔ اب کنویں کے وسط  سے ذرا ہٹ کر   ایک دیوار:
\begin{align*}
V (x) = f (t) \delta (x - \frac{a}{2} - \epsilon)
\end{align*}
  آہستہ آہستہ کھڑی کی جاتی ہے، جہاں \عددی{f (t)} آہستہ آہستہ \عددی{0}سے \عددی{\infty} تک بڑھتا ہے۔  مسئلہ حر ناگزر کے تحت،  یہ ذرہ ارتقائی  ہیملٹنی کے زمینی حال میں  رہے گا۔ 
\begin{enumerate}[a.]
\item
وقت \عددی{t \to \infty} پر زمینی حال تلاش کریں (اور اس  کا خاکہ بنائیں)۔ \ترچھا{ اشارہ:}  یہ اس لامتناہی چوکور کنویں کا زمینی حال ہوگا جس میں \عددی{a/2 + \epsilon} پر ناقابل گزر رکاوٹ ہو۔ آپ دیکھیں گے کہ ذرہ بائیں ہاتھ کے نسبتاً بڑے  حصے  میں رہنے کا پابند ہوگا۔ 
\item
وقت \عددی{ t} پر ہیملٹنی کے  زمینی حال کی ماورائی مساوات تلاش کریں۔ \ترچھا{ جواب:} 
\begin{align*}
z \sin z = T [\cos z - \cos (z \delta)]
\end{align*}
جہاں \عددی{z \equiv ka}،  \عددی{T \equiv maf (t)/\hslash^2}، \عددی{\delta \equiv 2 \epsilon/a}،  اور \عددی{k \equiv \sqrt{2m E} /\hslash} ہیں۔ 
\item
اب \عددی{\delta = 0} لیتے ہوئے \عددی{z} کے لیے ترسیمی طور پر حل کرکے دکھائیں کے \عددی{ T} کی قیمت \عددی{ 0} تا  \عددی{\infty} ہونے سے \عددی{z} کی قیمت \عددی{\pi} تا \عددی{2\pi} ہوگی۔  اس نتیجہ کی وضاحت پیش کریں۔
\item 
اب \عددی{\delta = 0.01} لیتے ہوئے \عددی{T = 0, 1, 5, 20, 100,1000}  کے لیے اعدادی طریقے سے   \عددی{z} حاصل کریں۔ 
\item
کنویں کے دائیں نصف حصہ میں ذرہ پائے جانے کا احتمال،  بطور \عددی{z} اور \عددی{\delta} کا تفاعل،  تلاش کریں۔ \ترچھا{ جواب:}

   \عددی{P_r = 1/[1 + (I_{+} / I_{-})]} 
جہاں \عددی{I_{\pm} \equiv [1 \pm \delta - (1/z) \sin (z(1\pm \delta))] \sin^2 [z (1 \mp \delta)/2]} ہوگا۔  جزو-د  میں دیے گئے \عددی{T}  کے لئے اس ریاضی  جملے  کی قیمتیں تلاش کریں۔  اپنے نتائج پر تبصرہ کریں۔ 
\item
\عددی{T} اور \عددی{\delta} کی انہی قیمتوں کے لئے زمینی حال تفاعل موج ترسیم کریں۔  آپ دیکھیں گے کہ رکاوٹ بلند ہونے سے کس طرح ذرہ کنویں کے بائیں نصف حصہ میں رہنے کا پابند ہو جاتا ہے۔ 
\end{enumerate}

\ابتدا{سوال}\شناخت{سوال_حر_جبری_مرتعش}
فرض کریں یک بُعدی ہارمونی مرتعش  (کمیت \عددی{m} ، تعدد \عددی{\omega})  پر \عددی{F(t) = m\omega^2 f (t)}،  جہاں \عددی{f(t)} کوئی مخصوص تفاعل ہے،  کی  جبری قوت اثر انداز ہوتی ہے ( میں نے \عددی{m \omega^2} کو صریحاً  لکھا ہے؛  \عددی{f(t)} کا بُعد فاصلہ ہے)۔  اس کی  ہیملٹنی درج ذیل ہوگی۔
\begin{align}
H(t) = - \frac{\hslash^2}{2m} \frac{\partial^2}{\partial x^2} + \frac{1}{2} m \omega^2 x^2 - m \omega^2 x f (t)
\end{align}
فرض کریں وقت \عددی{t = 0} پر یہ قوت پہلی مرتبہ چالو کی جاتی ہے:  لہٰذا \عددی{t \le 0} پر \عددی{f(t) = 0} ہوگا۔  اس نظام کو کلاسیکی  اور کوانٹائی میکانیات دونوں میں بالکل ٹھیک حل کیا جا سکتا ہے۔ 
\begin{enumerate}[a.]
\item
اگر مرتعش مبدا پر ساکن حال \عددی{(x_c (0) = \dot{x}_c (0) = 0)} سے آغاز کرے،  تب مرتعش کا کلاسیکی مقام کیا ہوگا۔ \ترچھا{ جواب:} 
\begin{align}\label{مساوات_حر_کلاسیکی_مقام}
x_c (t) = \omega \int_0^{t} f(t') \sin [\omega (t - t')] \dif t'
\end{align}
\item
جبری  قوت کی غیر موجودگی میں،  اگر مرتعش \عددی{ n} ویں  حال ( \عددی{\Psi (x, 0) = \psi_n (x)} جہاں \عددی{\psi_n(x)} مساوات  \حوالہ{مساوات_شروڈنگر_ہارمونی_حالات}  دیتی ہے)  سے آغاز کرے،  تو  دکھائیں کہ ( تابع وقت)  مساوات شروڈنگر کے حل کو درج ذیل لکھا جا سکتا ہے۔ 
\begin{align}
\Psi (x, t) = \psi_n (x - x_c) e^{\frac{i}{\hslash} [- (n + \frac{1}{2}) \hslash \omega t + m \dot{x}_c (x - \frac{x_c}{2}) + \frac{m \omega^2}{2} \int_0^t f (t') x_c (t')  \dif t']}
\end{align}
\item
دکھائیں  کہ \عددی{H(t)} کے امتیازی تفاعلات اور امتیازی اقدار درج ذیل ہونگے۔ 
\begin{align}
\psi_n (x, t) = \psi_n (x - f); \quad E_n (t) = \big ( n + \frac{1}{2} \big ) \hslash \omega - \frac{1}{2} m \omega^2 f^2
\end{align}
\item
دکھائیں کہ حر ناگزر تخمین کی صورت میں کلاسیکی مقام  (مساوات  \حوالہ{مساوات_حر_کلاسیکی_مقام}) سادہ روپ:  \عددی{x_c (t) \cong f (t)} اختیار کرتی ہے۔موجودہ سیاق و سباق کے لحاظ سے، حر ناگزر پن  تفاعل \عددی{ f} کے  وقتی تفرق پر کیا پابندی عائد کرتا ہے۔  \ترچھا{اشارہ:}  \عددی{\sin [\omega (t - t')]} کو \عددی{(1/\omega ) (\dif/{\dif t'}) \cos[\omega (t - t')]} لکھ کر تکمل بالحصص استعمال کریں۔
\item
اس مثال کے لیے مسئلہ حر ناگزر کی تصدیق جزو-ج  اور  جزو-د کے نتائج سے درج ذیل دکھا کر کریں ۔
\begin{align}
\Psi (x, t) \cong \psi_n (x, t) e^{i \theta_n (t)} e^{i \gamma_n (t)}
\end{align}
تصدیق کریں  کہ حرکی ہیّت کا روپ درست ہے  (مساوات  \حوالہ{مساوات_حر_درست_روپ})۔   کیا ہندسی ہیّت آپ کی  توقعات کے مطابق ہے ؟
\end{enumerate}

حر ناگزر تخمین کو مساوات  \حوالہ{مساوات_حر_مساوات_شروڈنگر_ب}  میں عددی سر \عددی{c_m(t)} کے  \اصطلاح{حر ناگزر تسلسل}\فرہنگ{حر ناگزر تسلسل}\حاشیہب{adiabatic series}\فرہنگ{adiabatic series}  کا پہلا جزو تصور کیا جا سکتا ہے۔ فرض کریں  نظام \عددی{ n} ویں حال سے آغاز کرتا ہے؛ حر ناگزر تخمین میں،    اضافی تابع وقت ہندسی ہیّتی جزو ضربی  (مساوات  \حوالہ{مساوات_حر_سی_ایم})     حاصل کرنے کے علاوہ، یہ  \عددی{ n} وی حال میں ہی رہتا ہے۔
\begin{align*} 
c_m (t) = \delta_{mn} e^{i \gamma_n (t)}
\end{align*} 
\begin{enumerate}[a.]
\item
اس کو مساوات\حوالہ{مساوات_حر_تفرق_عددی_سر}کے دائیں ہاتھ میں پُر کر کے حر ناگزر کی  \قول{ پہلی تصحیح} حاصل کریں۔
\begin{align}\label{مساوات_حر_آخری_سوال}
c_m (t) = c_m (0) - \int_0^t \big\langle \psi_m (t') \big | \frac{\partial}{\partial t'} \psi_n (t') \big \rangle e^{i \gamma_n (t')} e^{i (\theta_n (t') - \theta_m (t'))} \dif t'
\end{align}
یہ  ہمیں \ترچھا{ قریب}  حر ناگزر  طریق  میں تحویلی احتمالات کا حساب کر نے کے  قابل بناتا ہے۔\قول{ دوسری تصحیح}  کی خاطر ہم مساوات \حوالہ{مساوات_حر_آخری_سوال}کو مساوات\حوالہ{مساوات_حر_تفرق_عددی_سر}کے دائیں ہاتھ میں پُر کریں گے،  وغیرہ،  وغیرہ۔ 
\item
ایک مثال کے طور پر،  مساوات \حوالہ{مساوات_حر_آخری_سوال}کا اطلاق جبری مرتعش  (سوال \حوالہ{سوال_حر_جبری_مرتعش})   پر کریں۔  دکھائیں کہ (قریب حر ناگزر تخمین میں)  صرف قریبی دو  سطحوں، جن کے لیے درج ذیل ہوگا، میں تحویل ممکن ہے۔ 
\begin{align*}
c_{n + 1} (t)& = i \sqrt{\frac{m \omega}{2 \hslash}} \sqrt{n + 1} \int_0^t \dot{f} (t') e^{i \omega t'} \dif t' \\
c_{n - 1} (t) &= i \sqrt{\frac{m \omega}{2 \hslash}} \sqrt{n} \int_0^t \dot{f} (t') e^{-i \omega t'} \dif t'
\end{align*}
(یقیناً،  تحویلی  \ترچھا{احتمالات} ان کے مطلق مربع کے برابر ہوں گے۔) 
\end{enumerate}
